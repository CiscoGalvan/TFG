\chapter*{Dedicatoria}

% Primera dedicatoria a la derecha
\begin{flushright}
\begin{minipage}[c]{8.5cm}
\flushright{\textit{
A mi familia y amigos, por estar siempre ahí, por cada palabra de aliento y por no dejar que me rindiera cuando más lo necesitaba.\\
Y a mi abuela, cuya mirada sigue llenándome de fuerza, aunque ya no pueda decirlo con palabras.\\
A todos vosotros, este logro también es vuestro.
}}

\vspace{0.3cm} % Espacio entre el texto y el nombre
\flushright{\textbf{-- Cristina Mora Velasco}}
\end{minipage}
\end{flushright}

\vspace{1cm} % Espacio vertical entre las dedicatorias

}}

\vspace{0.3cm} % Espacio entre el texto y el nombre
\flushleft{\textbf{-- Francisco Miguel Galván Muñoz}}
\end{minipage}
